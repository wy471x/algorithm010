// 1. space swap time : time = O(n) space = O(n)
/************************************
public void moveZeroes(int[] nums) {
    int len = nums.length, zeroCount = 0;
    for (int i = 0; i < len; ++i) {
        if (nums[i] == 0) {
            zeroCount++;
        }
    }

    int[] ans = new int[len];
    int index = 0;
    for (int j = 0; j < len; ++j) {
        if (nums[j] != 0) {
            ans[index++] = nums[j];
        }
    }

    while(zeroCount != 0) {
        ans[index++] = 0;
        zeroCount--;
    }

    for (int k = 0; k < len; ++k) {
        nums[k] = ans[k];
    }

}

// @lc code=start
class Solution {
    public void moveZeroes(int[] nums) {
        int len = nums.length, lastNonZeroElementAt = 0;
        for (int cur = 0; cur < len; ++cur) {
            if (nums[cur] != 0) {
                nums[lastNonZeroElementAt++] = nums[cur];
            }
        }

        for (; lastNonZeroElementAt < len; ++lastNonZeroElementAt) {
            nums[lastNonZeroElementAt] = 0;
        }
    }
}
